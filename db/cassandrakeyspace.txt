CREATE KEYSPACE soundscale WITH REPLICATION = { 'class' : 'NetworkTopologyStrategy', 'replication_factor' : 3 }

CREATE TABLE songs(
   id int PRIMARY KEY,
   name text,
   plays text,
   likes int,
   reposts int
   );
